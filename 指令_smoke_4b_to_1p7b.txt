# ---------- 【填写区】路径与模型（已按 8x4090 机器填写） ----------
CODE_ROOT=/inspire/hdd/project/multi-agent/zhangweinan-24046/dk/CacheOPD/taopd-baseline

MEGATRON_LM_DIR=${CODE_ROOT}/Megatron-LM
CONDA_ENV=ta_opd                                  # 与 指令_环境准备_ta_opd.txt 创建的 env 名一致

OUTPUT_ROOT=${CODE_ROOT}/outputs/slime_opd
# -------------------------------------------------------

# 8 卡分配：teacher 占 7；actor 用 0,1（2 卡）；rollout 用 2（1 卡）；3,4,5,6 备用
export TEACHER_GPU=7

export TEACHER_MODEL STUDENT_HF STUDENT_TORCH_DIST PROMPT_DATA

# 环境激活（自动探测 conda 安装位置；若 conda 不在 PATH，请改用绝对路径）
source "$(conda info --base)/etc/profile.d/conda.sh"; conda activate "${CONDA_ENV}"   # 已全局关闭 nounset，conda 钩子不再报错

run_one () {
  local NAME="$1"; local MASK="$2"; local RATIO="$3"
  echo "===== SMOKE RUN: ${NAME} (mask=${MASK} ratio=${RATIO}) ====="
  RUN_NAME="smoke_${NAME}_$(date +%Y%m%d_%H%M%S)" \
  SAVE_DIR="${OUTPUT_ROOT}/${NAME}" \
  NUM_ROLLOUT=4 \
  ROLLOUT_BATCH_SIZE=4 \
  N_SAMPLES_PER_PROMPT=2 \
  GLOBAL_BATCH_SIZE=8 \
  ROLLOUT_MAX_RESPONSE_LEN=1024 \
  ROLLOUT_TEMPERATURE=1.0 \
  OPD_BUDGET_MASK="${MASK}" \
  OPD_BUDGET_RATIO="${RATIO}" \
  OPD_TOKEN_BANK_PAIR_ID=qwen3_4b_to_qwen3_1p7b \
  bash "${SLIME_DIR}/examples/on_policy_distillation/run-qwen3-1.7B-sampled-opd-sglang.sh"
  echo "===== DONE: ${NAME} -> ${OUTPUT_ROOT}/${NAME} ====="
}

# 1) Full OPD baseline：mask=none（全部 response token 都作直接 OPD 目标）
run_one pure_opd none 1.0

# 2) TA-OPD：mask=dlearn_high，10% 监督 token 预算
run_one ta_opd dlearn_high 0.10

echo "SMOKE 完成：检查两个 SAVE_DIR 下是否生成 latest_checkpointed_iteration.txt"
echo "  ${OUTPUT_ROOT}/pure_opd"
echo "  ${OUTPUT_ROOT}/ta_opd"
