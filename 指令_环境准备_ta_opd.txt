#!/usr/bin/env bash
# ============================================================
# 指令草稿（AI 辅助生成，需本人审阅 / 填写 / 认领后使用）
# 目标：搭建 TA-OPD 训练环境（适配 8 x 4090 / Ada sm89）
# 依据：TA-OPD/slime_ta_opd/docker/Dockerfile（已据此锁定版本；并去除 Hopper/GB 专属步骤）
# 用法：填好【填写区】-> bash 本文件；耗时较长（flash-attn/apex 源码编译，数十分钟）
# 遵循经验规则：本文件提交 git，远程 git pull 后执行
# ============================================================
set -eo pipefail
trap 'echo "[ABORT] 第 $LINENO 行命令失败，退出码 $?"' ERR   # 任意命令失败即打印位置，便于定位中断点

# ---------- 【填写区】路径（已按 8x4090 机器填写） ----------
CODE_ROOT=/inspire/hdd/project/multi-agent/zhangweinan-24046/dk/CacheOPD/taopd-baseline
TAOPD_DIR=${CODE_ROOT}/TA-OPD                 # 已克隆的 TA-OPD 仓库
SLIME_DIR=${TAOPD_DIR}/slime_ta_opd           # 完整 slime 树（含 train.py / setup.py / docker/patch）
MEGATRON_LM_DIR=${CODE_ROOT}/Megatron-LM      # 本脚本会在此 clone + 打 patch
OUTPUT_ROOT=${CODE_ROOT}/outputs/slime_opd    # 训练产物放代码目录内（已被 .gitignore 排除）
# -------------------------------------------------

# CUDA / torch 基准（必须一致；4090 用 CUDA 12.4 + torch 2.5.x 较稳）
# 若 sglang 装不上 cu124 的 sgl-kernel wheel，把下面两套都改成 cu121 / torch 2.4.0 再试。
CUDA_SHORT=124            # 用于 pip torch / sgl-kernel 的 cu 后缀
TORCH_VER=2.5.1
CONDA_ENV=ta_opd          # 环境名；与后面 smoke 指令的 CONDA_ENV 保持一致
PYTHON_VER=3.10

ENV_LOG=${OUTPUT_ROOT}/env_build_$(date +%Y%m%d_%H%M%S).log
mkdir -p "${OUTPUT_ROOT}"
exec > >(tee -a "${ENV_LOG}") 2>&1
echo "[$(date '+%F %T')] ENV BUILD START -> log=${ENV_LOG}"

# 4090 = Ada sm89：限定编译架构，避免 flash-attn/apex 全架构编译或报错
export TORCH_CUDA_ARCH_LIST="8.9"
export MAX_JOBS=16
export PIP_NO_CACHE_DIR=1

# ===================== PHASE 0: conda 环境 =====================
echo "===== PHASE 0: 创建 conda 环境 ${CONDA_ENV} (py${PYTHON_VER}) ====="
source "$(conda info --base)/etc/profile.d/conda.sh"
# 安全网：conda 的 activate.d/deactivate.d 钩子会引用若干未定义变量，先置空避免（即便未开 nounset 也稳住）
export _CONDA_PYTHON_SYSCONFIGDATA_NAME_USED="${_CONDA_PYTHON_SYSCONFIGDATA_NAME_USED:-}"
conda create -y -n "${CONDA_ENV}" python=${PYTHON_VER}
conda activate "${CONDA_ENV}"   # 先激活：后续 nvcc/pip/torch 全部装进 ta_opd 环境（已全局关闭 nounset，conda 钩子不再报错）

# --- CUDA 工具链：优先用机器自带系统 CUDA，否则 pip 装进当前(ta_opd)环境（跳过 conda 的 nvidia 频道） ---
NVCC_BIN=""
for cand in "${CUDA_HOME:-}/bin/nvcc" /usr/local/cuda/bin/nvcc "$(command -v nvcc 2>/dev/null)"; do
  [ -x "$cand" ] && NVCC_BIN="$cand" && break
done
if [ -n "$NVCC_BIN" ]; then
  export CUDA_HOME="$(cd "$(dirname "$(dirname "$NVCC_BIN")")" && pwd)"
  NV_VER="$(nvcc --version | grep -o 'release [0-9.]*' | head -1)"
  echo "[OK] 使用系统 CUDA: ${CUDA_HOME} (${NV_VER})"
  echo "[INFO] 已跳过 pip 安装 cuda-nvcc"
  if echo "$NV_VER" | grep -q 'release 1[0-1]\.'; then
    echo "[WARN] 系统 CUDA 版本 <12，可能和 torch cu124 不匹配；如编译 flash-attn/apex 报错请改用 CUDA 12.x"
  fi
else
  echo "[WARN] 未找到系统 nvcc，回退 pip 安装 nvidia-cuda-nvcc-cu12（轻量，不触发 conda 钩子）"
  pip install nvidia-cuda-nvcc-cu12 nvidia-cuda-runtime-cu12 nvidia-cuda-cupti-cu12
  # nvidia.cuda_nvcc 是命名空间包，__file__ 为 None，改用 site-packages 目录定位 nvcc
  SITE="$(python -c 'import sysconfig;print(sysconfig.get_path("purelib"))')"
  NVCC_BIN="$(ls "$SITE"/nvidia/cuda_nvcc/bin/nvcc 2>/dev/null | head -1)"
  [ -z "$NVCC_BIN" ] && NVCC_BIN="$(find "$SITE"/nvidia -name nvcc -type f 2>/dev/null | head -1)"
  export CUDA_HOME="$(cd "$(dirname "$(dirname "$NVCC_BIN")")" && pwd)"
  # 让编译器/链接器也能找到运行时头文件与库（来自 nvidia-cuda-runtime-cu12）
  export CPATH="${CUDA_HOME}/include:${SITE}/nvidia/cuda_runtime/include${CPATH:+:$CPATH}"
  export LIBRARY_PATH="${CUDA_HOME}/lib64:${CUDA_HOME}/lib:${SITE}/nvidia/cuda_runtime/lib${LIBRARY_PATH:+:$LIBRARY_PATH}"
  export LD_LIBRARY_PATH="${SITE}/nvidia/cuda_runtime/lib${LD_LIBRARY_PATH:+:$LD_LIBRARY_PATH}"
fi
export PATH="${CUDA_HOME}/bin:${PATH}"
echo "CUDA_HOME=${CUDA_HOME}  nvcc=$(command -v nvcc)"

# ===================== PHASE 1: torch + sglang（基准） =====================
echo "===== PHASE 1: torch ${TORCH_VER}+cu${CUDA_SHORT} + sglang 0.5.10.post1 ====="
pip install "torch==${TORCH_VER}" --index-url "https://download.pytorch.org/whl/cu${CUDA_SHORT}"
pip install "sglang==0.5.10.post1"
python -c "import torch, sglang; print('torch', torch.__version__, 'cuda', torch.version.cuda); print('sglang', sglang.__version__)"
echo "[CHECK] 若 sglang import 报 sgl-kernel wheel 缺失：把 CUDA_SHORT/TORCH_VER 改 cu121/torch2.4.0 重装 PHASE1"

# ===================== PHASE 2: Megatron-LM（锁 commit + 打 patch） =====================
echo "===== PHASE 2: clone Megatron-LM @1dcf0daf + 应用 megatron.patch ====="
if [ ! -d "${MEGATRON_LM_DIR}" ]; then
  git clone https://github.com/NVIDIA/Megatron-LM.git "${MEGATRON_LM_DIR}" --recursive
fi
cd "${MEGATRON_LM_DIR}"
git fetch --all
git checkout 1dcf0dafa884ad52ffb243625717a3471643e087
git submodule update --init --recursive
# 应用 TA-OPD 必需的 Megatron patch（slime 的 OPD 代码依赖它）
PATCH=${SLIME_DIR}/docker/patch/latest/megatron.patch
if [ -f "${PATCH}" ]; then
  git apply "${PATCH}" --3way || { echo "megatron.patch 应用失败，请手动解决冲突"; exit 1; }
  echo "[OK] applied ${PATCH}"
else
  echo "未找到 ${PATCH}，请确认 SLIME_DIR 指向 slime_ta_opd"
  exit 1
fi
pip install -e . --no-deps
echo "[CHECK] import megatron.core"
python -c "import megatron.core; print('megatron.core OK')"

# ===================== PHASE 3: 编译型依赖（慢） =====================
echo "===== PHASE 3: flash-attn 2.7.4.post1 + apex（源码编译，耐心等） ====="
# 仅装 FA2；**不要**装 Dockerfile 里的 flash-attention/hopper（FA3，Hopper 专属，4090 用不了）
pip install "flash-attn==2.7.4.post1" --no-build-isolation
pip install "git+https://github.com/NVIDIA/apex.git@10417aceddd7d5d05d7cbf7b0fc2daad1105f8b4" \
  --no-build-isolation --config-settings "--build-option=--cpp_ext" --config-settings "--build-option=--cuda_ext"

# ===================== PHASE 4: TransformerEngine 2.10（4090 可装 cu12x wheel） =====================
echo "===== PHASE 4: transformer_engine 2.10.0 ====="
pip install "transformer_engine[pytorch]==2.10.0" || \
  echo "[WARN] TE 安装失败；训练脚本已用 --transformer-impl local，前向不依赖 TE，但 Megatron import 可能需要它，请排查"

# ===================== PHASE 5: slime 依赖 + slime 自身 =====================
echo "===== PHASE 5: slime requirements + pip install -e . ====="
pip install -r "${TAOPD_DIR}/requirements.txt"
pip install -r "${SLIME_DIR}/requirements.txt"
cd "${SLIME_DIR}"
pip install -e . --no-deps
# slime 自带的一个 int4 kernel（密集 OPD 非必须，但 Dockerfile 装了，跳过也不影响 baseline）
( cd "${SLIME_DIR}/slime/backends/megatron_utils/kernels/int4_qat" && pip install . --no-build-isolation ) || \
  echo "[WARN] int4_qat kernel 安装失败（可选，baseline 可忽略）"

# ===================== PHASE 6: slime 的第三方 git 依赖 =====================
echo "===== PHASE 6: mbridge / Megatron-Bridge / torch_memory_saver / modelopt ====="
pip install "git+https://github.com/ISEEKYAN/mbridge.git@89eb10887887bc74853f89a4de258c0702932a1c" --no-deps
pip install "git+https://github.com/radixark/Megatron-Bridge.git@bridge" --no-deps --no-build-isolation
pip install "git+https://github.com/fzyzcjy/torch_memory_saver.git@d64a639" --no-cache-dir --force-reinstall
pip install "nvidia-modelopt[torch]>=0.37.0" --no-build-isolation
# flash-linear-attention 0.4.1（可选，MoE/特定特性用）
pip install "flash-linear-attention==0.4.1" || echo "[WARN] flash-linear-attention 可选"
# tilelang 的 cu128 wheel 仅当 CUDA==12.8 才匹配；不符则跳过
# pip install tilelang -f https://tile-ai.github.io/whl/nightly/cu128/ || echo "[WARN] tilelang 跳过（CUDA 不匹配）"

# numpy 必须 <2（Megatron 要求）
pip install "numpy<2"

# ===================== PHASE 7: 最终验证 =====================
echo "===== PHASE 7: 验证导入 + GPU ====="
python - <<'PY'
import importlib, sys
mods = ["torch","sglang","ray","megatron.core","transformer_engine","flash_attn",
        "slime","slime_plugins","apex","mbridge"]
for m in mods:
    try:
        importlib.import_module(m)
        print(f"  [OK] {m}")
    except Exception as e:
        print(f"  [FAIL] {m}: {e}")
import torch
print("torch", torch.__version__, "cuda avail:", torch.cuda.is_available(),
      "device count:", torch.cuda.device_count())
PY
nvidia-smi --query-gpu=index,name,memory.total --format=csv

echo "===== ENV BUILD DONE ====="
echo "SLIME_DIR    = ${SLIME_DIR}"
echo "MEGATRON_LM  = ${MEGATRON_LM_DIR}"
echo "CONDA_ENV    = ${CONDA_ENV}  (smoke 指令里 CONDA_ENV 请保持一致)"
echo "下一步：用 指令_权重转换_*.txt 转 student 权重，再跑 指令_smoke_4b_to_1p7b.txt"
